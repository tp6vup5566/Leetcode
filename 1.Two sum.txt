class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int,int> out;
        for(int i=0;i<nums.size();i++){
            out[nums[i]]=i;
        }
        for(int i=0;i<nums.size();i++){
            int left=target-nums[i];
            if(out.count(left) && out[left]!=i){
                return {i,out[left]};
            }
        }
        return {};
    }
};
